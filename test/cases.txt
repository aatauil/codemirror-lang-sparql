# Namespace

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
SELECT * WHERE {
  ?sub ?pred ?obj .
} LIMIT 10

==>

Unit(
  Query(
    Prologue(
      PrefixDecl(
        PREFIX
        Pname_ns,
        IriRef
      )
    )
    SelectQuery(
      SelectClause (
        SELECT
      )
      WhereClause(
        WHERE
        GroupGraphPattern(
          GroupGraphPatternSub (
            TriplesBlock (
              TriplesSameSubjectPath (
                VarOrTerm (
                  Var
                )
                PropertyListPathNotEmpty (
                  VerbSimple (
                    Var
                  )
                  ObjectListPath (
                    ObjectPath (
                      GraphNodePath (
                        VarOrTerm (
                          Var
                        )
                      )
                      AnnotationPath
                    )
                  )
                )
              )
            )
          )
        )
      )
      SolutionModifier (
        LimitOffsetClauses (
          LimitClause (
            LIMIT
            Integer
          )
        )
      )
    )
    ValuesClause
  )
)

# parse "a" as PathPrimary instead of parse error

SELECT * WHERE {
  ?sub a ?aClass .
}

==>

Unit(
  Query(
    Prologue,
    SelectQuery(
      SelectClause(
        SELECT
      ),
      WhereClause(
        WHERE,
        GroupGraphPattern(
          GroupGraphPatternSub(
            TriplesBlock(
              TriplesSameSubjectPath(
                VarOrTerm(
                  Var
                ),
                PropertyListPathNotEmpty(
                  VerbPath(
                    Path(
                      PathAlternative(
                        PathSequence(
                          PathEltOrInverse(
                            PathElt(
                              PathPrimary(
                                A_predicate
                              )
                            )
                          )
                        )
                      )
                    )
                  ),
                ObjectListPath(
                  ObjectPath(
                    GraphNodePath(
                      VarOrTerm(
                        Var
                      )
                    ),
                    AnnotationPath
                  )
                )
              )
            ),
            "."
          )
        )
      )
    ),
    SolutionModifier),
  ValuesClause))

# SPARQL 1.2 VERSION declaration

VERSION "1.2"
SELECT * { ?s ?p ?o }

==>

Unit(Query(Prologue(VersionDecl(VERSION,String)),SelectQuery(SelectClause(SELECT),WhereClause(GroupGraphPattern(GroupGraphPatternSub(TriplesBlock(TriplesSameSubjectPath(VarOrTerm(Var),PropertyListPathNotEmpty(VerbSimple(Var),ObjectListPath(ObjectPath(GraphNodePath(VarOrTerm(Var)),AnnotationPath)))))))),SolutionModifier),ValuesClause))

# SPARQL 1.2 Triple Term

SELECT * { <<( :s :p :o )>> ?p ?o }

==>

Unit(Query(Prologue,SelectQuery(SelectClause(SELECT),WhereClause(GroupGraphPattern(GroupGraphPatternSub(TriplesBlock(TriplesSameSubjectPath(VarOrTerm(TripleTerm(Iri(PrefixedName(Pname_ln)),Verb(VarOrIri(Iri(PrefixedName(Pname_ln)))),Iri(PrefixedName(Pname_ln)))),PropertyListPathNotEmpty(VerbSimple(Var),ObjectListPath(ObjectPath(GraphNodePath(VarOrTerm(Var)),AnnotationPath)))))))),SolutionModifier),ValuesClause))

# SPARQL 1.2 Reified Triple

SELECT * { << :s :p :o >> ?p ?o }

==>

Unit(Query(Prologue,SelectQuery(SelectClause(SELECT),WhereClause(GroupGraphPattern(GroupGraphPatternSub(TriplesBlock(TriplesSameSubjectPath(ReifiedTripleBlockPath(ReifiedTriple(Iri(PrefixedName(Pname_ln)),Verb(VarOrIri(Iri(PrefixedName(Pname_ln)))),Iri(PrefixedName(Pname_ln))),PropertyListPath(PropertyListPathNotEmpty(VerbSimple(Var),ObjectListPath(ObjectPath(GraphNodePath(VarOrTerm(Var)),AnnotationPath)))))))))),SolutionModifier),ValuesClause))

# SPARQL 1.2 Annotation

SELECT * { ?s :p ?o {| :source :wiki |} }

==>

Unit(Query(Prologue,SelectQuery(SelectClause(SELECT),WhereClause(GroupGraphPattern(GroupGraphPatternSub(TriplesBlock(TriplesSameSubjectPath(VarOrTerm(Var),PropertyListPathNotEmpty(VerbPath(Path(PathAlternative(PathSequence(PathEltOrInverse(PathElt(PathPrimary(Iri(PrefixedName(Pname_ln))))))))),ObjectListPath(ObjectPath(GraphNodePath(VarOrTerm(Var)),AnnotationPath(AnnotationBlockPath(PropertyListPathNotEmpty(VerbPath(Path(PathAlternative(PathSequence(PathEltOrInverse(PathElt(PathPrimary(Iri(PrefixedName(Pname_ln))))))))),ObjectListPath(ObjectPath(GraphNodePath(VarOrTerm(GraphTerm(Iri(PrefixedName(Pname_ln))))),AnnotationPath))))))))))))),SolutionModifier),ValuesClause))

# SPARQL 1.2 Language Direction

Unit(Query(Prologue,SelectQuery(SelectClause(SELECT),WhereClause(GroupGraphPattern(GroupGraphPatternSub(TriplesBlock(TriplesSameSubjectPath(VarOrTerm(Var),PropertyListPathNotEmpty(VerbPath(Path(PathAlternative(PathSequence(PathEltOrInverse(PathElt(PathPrimary(Iri(PrefixedName(Pname_ln))))))))),ObjectListPath(ObjectPath(GraphNodePath(VarOrTerm(GraphTerm(RDFLiteral(String,Langtag)))),AnnotationPath)))))))),SolutionModifier),ValuesClause))
